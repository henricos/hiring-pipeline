```typescript
"use client";

import { useState } from "react";
import {
  Accordion,
  AccordionContent,
  AccordionItem,
  AccordionTrigger,
} from "@/components/ui/accordion";
import { getVagasForDate } from "@/app/actions/research";
import type { Research } from "@/lib/repositories/research-repository";

interface ProfileDetailVagasProps {
  researches: Research[];
  profileId: string;
}

export function ProfileDetailVagas({
  researches,
  profileId,
}: ProfileDetailVagasProps) {
  const [selectedDate, setSelectedDate] = useState<string | null>(
    researches[0]?.date ?? null
  );
  const [expandedVagas, setExpandedVagas] = useState<Record<string, any>>({});

  // Empty state
  if (researches.length === 0) {
    return (
      <div className="text-center py-12">
        <h3 className="text-title-md font-medium text-on-surface mb-2">
          Nenhuma pesquisa de mercado
        </h3>
        <p className="text-body-md text-on-surface/60 mb-4">
          Nenhuma pesquisa foi realizada para este perfil.
        </p>
        <p className="text-body-md text-on-surface/60">
          Execute a skill{" "}
          <code className="bg-surface-container px-2 py-1 rounded-sm">
            /pesquisar-mercado
          </code>{" "}
          para gerar dados.
        </p>
      </div>
    );
  }

  const selectedResearch = researches.find((r) => r.date === selectedDate);

  // Função para carregar vagas quando accordion abre
  const handleVagasLoad = async (date: string) => {
    if (!expandedVagas[date]) {
      const vagas = await getVagasForDate(profileId, date);
      setExpandedVagas((prev) => ({ ...prev, [date]: vagas }));
    }
  };

  const vagasData = selectedDate ? expandedVagas[selectedDate] : null;
  const jobsList = vagasData?.jobs || [];

  return (
    <div className="space-y-6">
      {/* Seletor de data (múltiplas pesquisas) */}
      {researches.length > 1 && (
        <div className="flex items-center gap-4">
          <label className="text-label-sm uppercase text-on-surface/60">
            Pesquisa:
          </label>
          <select
            value={selectedDate ?? ""}
            onChange={(e) => {
              setSelectedDate(e.target.value);
              handleVagasLoad(e.target.value);
            }}
            className="rounded-sm bg-surface-container px-3 py-2 text-body-md border border-outline-variant"
          >
            {researches.map((r) => (
              <option key={r.date} value={r.date}>
                {r.date}
              </option>
            ))}
          </select>
        </div>
      )}

      {/* Lista de pesquisas com datas */}
      <div className="space-y-2">
        {researches.map((research) => (
          <div
            key={research.date}
            className="flex items-center justify-between py-3 px-4 rounded-sm bg-surface-container-low hover:bg-surface-container cursor-pointer transition-colors"
            onClick={() => {
              setSelectedDate(research.date);
              handleVagasLoad(research.date);
            }}
          >
            <div className="flex-1">
              <p className="text-body-md font-medium text-on-surface">
                {research.date}
              </p>
            </div>
            <span className="text-label-sm text-on-surface/60">
              Pesquisa disponível
            </span>
          </div>
        ))}
      </div>

      {/* Accordion de vagas para pesquisa selecionada */}
      {selectedResearch && jobsList.length > 0 && (
        <div className="space-y-4">
          <h4 className="text-title-md font-medium text-on-surface">
            Vagas encontradas
          </h4>
          <Accordion type="single" collapsible>
            {jobsList.map((job, idx) => (
              <AccordionItem key={idx} value={`job-${idx}`}>
                <AccordionTrigger className="text-body-md hover:no-underline">
                  <div className="flex items-center gap-3 text-left">
                    <span className="font-medium">{job.title}</span>
                    <span className="text-on-surface/60">{job.company}</span>
                  </div>
                </AccordionTrigger>
                <AccordionContent>
                  <div className="space-y-3 pt-4">
                    <p className="text-body-md">
                      <strong>Empresa:</strong> {job.company} ({job.companySize || "porte desconhecido"})
                    </p>
                    {job.stack && job.stack.length > 0 && (
                      <div>
                        <p className="text-body-md font-medium mb-2">Stack:</p>
                        <div className="flex flex-wrap gap-2">
                          {job.stack.map((tech, i) => (
                            <span
                              key={i}
                              className="px-2 py-1 rounded-sm bg-surface-container text-label-sm text-on-surface/80"
                            >
                              {tech}
                            </span>
                          ))}
                        </div>
                      </div>
                    )}
                    {job.snippet && (
                      <p className="text-body-md text-on-surface/80">
                        {job.snippet}
                      </p>
                    )}
                  </div>
                </AccordionContent>
              </AccordionItem>
            ))}
          </Accordion>
        </div>
      )}

      {selectedResearch && jobsList.length === 0 && (
        <p className="text-body-md text-on-surface/60 text-center py-8">
          Nenhuma vaga encontrada para {selectedDate}.
        </p>
      )}
    </div>
  );
}
```

```typescript
"use client";

import type { Research } from "@/lib/repositories/research-repository";

const SECTION_HEADING_CLASS =
  "text-[1.125rem] font-medium text-on-surface mb-4";

interface ProfileDetailResumoProps {
  researches: Research[];
}

// Type para -resumo.json (simplificado — pode ser more specific later)
interface ResearchSummary {
  commonTitles?: string[];
  titleAliases?: string[];
  stackFrequency?: Record<string, number>;
  salaryRange?: { min: number; max: number } | null;
  salaryGuide?: {
    min: number;
    max: number;
    currency?: string;
    sources?: Array<{ portal: string; year: number }>;
  };
  emergingStack?: string[];
  commonBehaviors?: string[];
  commonChallenges?: string[];
  archetypes?: Array<{ name: string; count?: number }>;
  profileHints?: {
    responsibilities?: string[];
    qualifications?: Array<{ text: string; required: boolean }>;
    behaviors?: string[];
    challenges?: string[];
  };
}

export function ProfileDetailResumo({
  researches,
}: ProfileDetailResumoProps) {
  // Empty state
  if (researches.length === 0) {
    return (
      <div className="text-center py-12">
        <h3 className="text-title-md font-medium text-on-surface mb-2">
          Nenhum resumo de mercado
        </h3>
        <p className="text-body-md text-on-surface/60">
          Execute a skill{" "}
          <code className="bg-surface-container px-2 py-1 rounded-sm">
            /pesquisar-mercado
          </code>{" "}
          para gerar dados.
        </p>
      </div>
    );
  }

  // Usar pesquisa mais recente (primeiro elemento — listByProfileId retorna DESC)
  const resumo = researches[0] as unknown as ResearchSummary;

  return (
    <div className="space-y-8">
      {/* Data do resumo (se múltiplas pesquisas) */}
      {researches.length > 1 && (
        <p className="text-label-sm text-on-surface/60">
          Resumo de: {researches[0].date}
        </p>
      )}

      {/* Stack Frequency — RANQUEADO por contagem decrescente */}
      {resumo.stackFrequency && Object.keys(resumo.stackFrequency).length > 0 && (
        <section>
          <h3 className={SECTION_HEADING_CLASS}>Stack Frequência</h3>
          <div className="space-y-2">
            {Object.entries(resumo.stackFrequency)
              .sort(([, a], [, b]) => (b as number) - (a as number))
              .map(([tech, count]) => (
                <div
                  key={tech}
                  className="flex items-center justify-between py-2 px-3 rounded-sm bg-surface-container-low"
                >
                  <span className="text-body-md text-on-surface">{tech}</span>
                  <span className="text-label-sm font-medium text-on-surface/70">
                    {count} menções
                  </span>
                </div>
              ))}
          </div>
        </section>
      )}

      {/* Faixa Salarial com atribuição de fontes */}
      {resumo.salaryGuide && (
        <section>

      {resumo.commonBehaviors && resumo.commonBehaviors.length > 0 && (
        <section>
          <h3 className={SECTION_HEADING_CLASS}>Comportamentos Comuns</h3>
          <ul className="space-y-1">
            {resumo.commonBehaviors.map((behavior, idx) => (
              <li
                key={idx}
                className="text-body-md text-on-surface flex gap-3"
              >
                <span className="shrink-0">•</span>
                <span>{behavior}</span>
              </li>
            ))}
          </ul>
        </section>
      )}

      {/* Desafios Comuns */}
      {resumo.commonChallenges && resumo.commonChallenges.length > 0 && (
        <section>
          <h3 className={SECTION_HEADING_CLASS}>Desafios Comuns</h3>
          <ul className="space-y-1">
            {resumo.commonChallenges.map((challenge, idx) => (
              <li
                key={idx}

                <span className="shrink-0">•</span>

        </section>

      {/* Arquétipos — ranqueado por contagem */}
      {resumo.archetypes && resumo.archetypes.length > 0 && (
        <section>
          <h3 className={SECTION_HEADING_CLASS}>Arquétipos</h3>
          <ul className="space-y-1">
            {resumo.archetypes
              .sort((a, b) => (b.count || 0) - (a.count || 0))
              .map((arch, idx) => (
                <li key={idx} className="text-body-md text-on-surface flex gap-3">
                  <span className="shrink-0">•</span>
                  <span>
                    {arch.name} ({arch.count || 0} menções)
                  </span>
                </li>
              ))}
          </ul>
        </section>
      )}

      {/* Profile Hints — Responsabilidades sugeridas */}
      {resumo.profileHints?.responsibilities &&
        resumo.profileHints.responsibilities.length > 0 && (
          <section>
            <h3 className={SECTION_HEADING_CLASS}>
              Responsabilidades Sugeridas
            </h3>
            <ul className="space-y-1">
              {resumo.profileHints.responsibilities.map((item, idx) => (
                <li
                  key={idx}
                  className="text-body-md text-on-surface flex gap-3"
                >
                  <span className="shrink-0">•</span>
                  <span>{item}</span>
                </li>
              ))}
            </ul>
          </section>
        )}

      {/* Profile Hints — Qualificações sugeridas */}
      {resumo.profileHints?.qualifications &&
        resumo.profileHints.qualifications.length > 0 && (
          <section>
            <h3 className={SECTION_HEADING_CLASS}>
              Qualificações Sugeridas
            </h3>
            <div className="space-y-2">
              {resumo.profileHints.qualifications.map((qual, idx) => (
                <p key={idx} className="text-body-md text-on-surface">
                  {qual.text}
                  {qual.required && (
                    <span className="ml-2 text-label-sm text-on-surface/60">
                      (obrigatório)
                    </span>
                  )}
                </p>
              ))}
            </div>
          </section>
        )}
    </div>
  );
}
```
