```typescript
import AdmZip from "adm-zip";
import fs from "fs";
import path from "path";
import type { Vacancy } from "@/lib/vacancy";
import type { JobProfile } from "@/lib/profile";
import type { AreaSettings } from "@/lib/settings";

// XML escape function (per Pitfall 4: RESEARCH.md)
// Mandatory: & < > " '
export function escapeXml(value: string): string {
  return value
    .replace(/&/g, "&amp;")
    .replace(/</g, "&lt;")
    .replace(/>/g, "&gt;")
    .replace(/"/g, "&quot;")
    .replace(/'/g, "&apos;");
}

// Cell mapping based on RESEARCH.md § Mapeamento 1-a-1
// Cells where values are written in the template (sheet1.xml)
// Note: This is a simplified mapping. Per RESEARCH.md A2, exact cell addresses require
// a diagnostic script to inspect the template. Below are placeholder mappings based on
// the research document's suggested row/column pattern.
const CELL_MAPPING = {
  // Group 1: Profile data (inherited from JobProfile)
  title: "D6",                    // Cargo (título interno)
  suggestedTitle: "D8",           // Cargo sugerido para anúncio
  experienceLevel: "D10",         // Tempo de experiência
  educationLevel: "D12",          // Nível de escolaridade
  educationCourse: "D13",         // Curso
  postGraduateLevel: "D15",       // Pós-graduação
  englishLevel: "D18",            // Inglês
  spanishLevel: "D20",            // Espanhol
  // Multi-line fields (rows 43-58)
  responsibilities: "B43:C46",    // Responsabilidades (rows 43-46, merged cells)
  qualifications: "B47:C49",      // Requisitos (rows 47-49)
  behaviors: "B50:C53",           // Comportamentais (rows 50-53)
  challenges: "B54:C57",          // Principais desafios (rows 54-57)
  additionalInfo: "B58",          // Informações complementares
  systemsRequired: "D61",         // Sistemas necessários (as text, not checkboxes)
  networkFolders: "D68",          // Pastas de rede

  // Group 2: Vacancy-specific data
  requestType: "D26",             // Tipo de requisição
  quantity: "D27",                // Quantidade de vagas
  costCenter: "D28",              // Centro de custo
  salaryRange: "D29",             // Faixa salarial
  confidential: "D30",            // Vaga confidencial (text: Sim/Não)
  budgeted: "D31",                // Vaga orçada
  headcountIncrease: "D32",       // Aumento de quadro
  replacedPerson: "D33",          // Nome do substituído (if headcountIncrease=false)
  workSchedule: "D34",            // Horário de trabalho
  travelRequired: "D35",          // Disponibilidade para viagens (text: Sim/Não)
  workMode: "D36",                // Modalidade de trabalho
  expectedHireDate: "D37",        // Data prevista de contratação

  // Group 3: Common data (from AreaSettings)
  managerName: "D22",             // Vaga solicitada por
  godfather: "D23",               // Nome do padrinho
  immediateReport: "D24",         // Reporte imediato
  mediateReport: "D25",           // Reporte mediato
  teamComposition: "B59:C62",     // Composição da equipe (rows 59-62)
};

export function generateVacancyForm(
  templatePath: string,
  outputPath: string,
  vacancy: Vacancy,
  profile: JobProfile,
  settings: AreaSettings
): void {
  // Validate template exists
  if (!fs.existsSync(templatePath)) {
    throw new Error(`Template não encontrado: ${templatePath}`);
  }

  // Load template as ZIP
  const zip = new AdmZip(templatePath);

  // Extract sheet1.xml (where data is written)
  const sheetEntry = zip.getEntry("xl/worksheets/sheet1.xml");
  if (!sheetEntry) {
    throw new Error("sheet1.xml não encontrado no template");
  }

  let sheet1Xml = sheetEntry.getData().toString("utf-8");

  // Build cell values map (Group 1, 2, 3)
  const cellValues: Record<string, string> = {
    // Group 1: Profile
    [CELL_MAPPING.title]: profile.title,
    [CELL_MAPPING.suggestedTitle]: profile.suggestedTitle,
    [CELL_MAPPING.experienceLevel]: profile.experienceLevel,
    [CELL_MAPPING.educationLevel]: profile.educationLevel,
    [CELL_MAPPING.educationCourse]: profile.educationCourse || "",
    [CELL_MAPPING.postGraduateLevel]: profile.postGraduateLevel,
    [CELL_MAPPING.englishLevel]: profile.englishLevel,
    [CELL_MAPPING.spanishLevel]: profile.spanishLevel,
    [CELL_MAPPING.responsibilities]: profile.responsibilities,
    [CELL_MAPPING.qualifications]: profile.qualifications,
    [CELL_MAPPING.behaviors]: profile.behaviors,
    [CELL_MAPPING.challenges]: profile.challenges,
    [CELL_MAPPING.additionalInfo]: profile.additionalInfo,
    [CELL_MAPPING.systemsRequired]: profile.systemsRequired || "",
    [CELL_MAPPING.networkFolders]: profile.networkFolders || "",

    // Group 2: Vacancy
    [CELL_MAPPING.requestType]: vacancy.requestType,
    [CELL_MAPPING.quantity]: vacancy.quantity.toString(),
    [CELL_MAPPING.costCenter]: vacancy.costCenter,
    [CELL_MAPPING.salaryRange]: vacancy.salaryRange,
    [CELL_MAPPING.confidential]: vacancy.confidential ? "Sim" : "Não",
    [CELL_MAPPING.budgeted]: vacancy.budgeted ? "Sim" : "Não",
    [CELL_MAPPING.headcountIncrease]: vacancy.headcountIncrease ? "Sim" : "Não",
    [CELL_MAPPING.replacedPerson]: vacancy.replacedPerson || "",
    [CELL_MAPPING.workSchedule]: vacancy.workSchedule,
    [CELL_MAPPING.travelRequired]: vacancy.travelRequired ? "Sim" : "Não",
    [CELL_MAPPING.workMode]: vacancy.workMode,
    [CELL_MAPPING.expectedHireDate]: vacancy.expectedHireDate,

    // Group 3: Settings (common data)
    [CELL_MAPPING.managerName]: settings.managerName,
    [CELL_MAPPING.godfather]: settings.godfather,
    [CELL_MAPPING.immediateReport]: settings.immediateReport,
    [CELL_MAPPING.mediateReport]: settings.mediateReport,
    [CELL_MAPPING.teamComposition]: settings.teamComposition,
  };

  // Apply cell values via regex substitution (surgical edit)
  // For each cell, find pattern <c r="D6" s="XX"/> (empty cell) or <c r="D6"... (existing cell)
  // and replace with value in inlineStr format
  for (const [cellAddr, value] of Object.entries(cellValues)) {
    const escapedValue = escapeXml(value);

    // Pattern 1: Empty cell <c r="D6" s="59"/> → <c r="D6" s="59"><is><t>value</t></is></c>
    const emptyCellPattern = new RegExp(
      `<c r="${cellAddr.replace(/\$/g, "\\$")}" s="(\\d+)"/>`,
      "g"
    );
    sheet1Xml = sheet1Xml.replace(
      emptyCellPattern,
      `<c r="${cellAddr}" s="$1"><is><t>${escapedValue}</t></is></c>`
    );

    // Pattern 2: Existing cell with value (t="s" for shared string)
    // <c r="D6" s="59" t="s"><v>8</v></c> → <c r="D6" s="59"><is><t>value</t></is></c>
    const existingCellPattern = new RegExp(
      `<c r="${cellAddr.replace(/\$/g, "\\$")}"[^>]*(?:t="[^"]*")?[^>]*>.*?</c>`,
      "g"
    );
    sheet1Xml = sheet1Xml.replace(
      existingCellPattern,
      `<c r="${cellAddr}"><is><t>${escapedValue}</t></is></c>`
    );
  }

  // Update sheet1.xml in ZIP (preserves all other members: VML, ctrlProps, styles, etc.)
  zip.updateFile("xl/worksheets/sheet1.xml", Buffer.from(sheet1Xml, "utf-8"));

  // Ensure output directory exists
  const outputDir = path.dirname(outputPath);
  fs.mkdirSync(outputDir, { recursive: true });

  // Write ZIP to output path (preserves compression settings, file order, all members)
  zip.writeZip(outputPath);
}
```

Key points per RESEARCH.md:
- Do NOT use exceljs or xlsx (they discard VML/ctrlProps)
- escapeXml() mandatory for all values before inserting into XML
- Regex pattern handles both empty cells (<c r="D6" s="59"/>) and existing values
- zip.updateFile() + zip.writeZip() preserves all ZIP members except sheet1.xml (which is modified)
- CELL_MAPPING is a starting point; exact cell addresses require inspection of actual template (A2 in Assumptions Log)

import { describe, it, expect } from "vitest";
import fs from "fs";
import path from "path";
import {
  escapeXml,
  generateVacancyForm,
} from "@/lib/excel-generator";
import { createDefaultVacancy } from "@/lib/vacancy";
import { env } from "@/lib/env";

describe("Excel Generator", () => {
  describe("escapeXml", () => {
    it("escapes ampersand", () => {
      expect(escapeXml("P&D")).toBe("P&amp;D");
    });

    it("escapes less-than", () => {
      expect(escapeXml("<test>")).toBe("&lt;test&gt;");
    });

    it("escapes quotes", () => {
      expect(escapeXml('He said "hello"')).toBe('He said &quot;hello&quot;');
    });

    it("preserves regular text", () => {
      expect(escapeXml("Hello World")).toBe("Hello World");
    });
  });

  describe("generateVacancyForm", () => {
    const templatePath = path.join(env.DATA_PATH, "templates", "requisicao-de-pessoal.xlsx");
    const outputDir = path.join(env.DATA_PATH, "forms-test");

    beforeEach(() => {
      if (!fs.existsSync(outputDir)) {
        fs.mkdirSync(outputDir, { recursive: true });

<threat_model>
## Trust Boundaries

| Boundary | Description | Treatment |
|----------|-------------|-----------|
| Cell values → Excel XML | Profile, vacancy, and settings data written to cells | XML-escaped via escapeXml() before insertion (& < > " ' all encoded). Malformed XML prevented. |
| Template file (DATA_PATH/templates/) | Trusted (manager-controlled). If template substituted with malicious ZIP, generateVacancyForm could be exploited. | Accept: DATA_PATH is trusted deployment context. Mitigated by code review of template before deployment. |
| Output file location (DATA_PATH/forms/) | Excel file written to trusted data directory | Mitigated by path validation (no ".." or "/" in ID) — prevents path traversal. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-05 | Tampering | FormData XML injection into Excel via unescaped values | mitigate | escapeXml() mandatory for all values before XML insertion (task 1, escapeXml function) |
| T-03-06 | Tampering | CELL_MAPPING incorrect → values written to wrong cells | accept | Mapping is starting point per RESEARCH.md A2; requires diagnostic script to confirm exact cells. Implementation is correct given mapping. |
| T-03-08 | Denial of Service | Malformed Excel template causes zip.writeZip() hang | mitigate | Template validation: file must exist (checked before opening) and must be valid ZIP (AdmZip throws if corrupt). Clear error message. |

All threats mitigated or accepted per ASVS L1.
</threat_model>
