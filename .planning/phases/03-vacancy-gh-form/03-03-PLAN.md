```typescript
  profiles: JobProfile[];
  vacancy?: Vacancy;
  onSubmitAction: (prevState: { error?: string } | null, formData: FormData) => Promise<{ error?: string } | void>;
}

export function VacancyForm({ profiles, vacancy, onSubmitAction }: VacancyFormProps) {

  const [selectedProfileId, setSelectedProfileId] = useState(vacancy?.profileId ?? "");
  const [headcountIncrease, setHeadcountIncrease] = useState(vacancy?.headcountIncrease ?? false);

          variant="outline"
          onClick={() => router.back()}
          disabled={isPending}
        >
          Cancelar
        </Button>
      </div>
    </form>
  );
}
```

```typescript
"use client";

import React, { useState } from "react";
import Link from "next/link";
import { useRouter } from "next/navigation";
import { Button } from "@/components/ui/button";
import { Badge } from "@/components/ui/badge";
import {
  AlertDialog,
  AlertDialogAction,
  AlertDialogCancel,
  AlertDialogContent,
  AlertDialogDescription,
  AlertDialogHeader,
  AlertDialogTitle,
} from "@/components/ui/alert-dialog";
import { Pencil, Trash2, ChevronRight } from "lucide-react";
import type { Vacancy } from "@/lib/vacancy";
import type { JobProfile } from "@/lib/profile";
import { deleteVacancy, advanceVacancyStatus } from "@/app/actions/vacancy";

interface VacancyListProps {
  vacancies: Vacancy[];
  profiles: Map<string, JobProfile>; // For quick lookup of profile title
}

const STATUS_BADGE_VARIANT: Record<string, "default" | "secondary" | "destructive"> = {
  "Aberta": "default",        // blue
  "Em andamento": "secondary", // neutral gray
  "Encerrada": "destructive",  // red/destructive
};

export function VacancyList({ vacancies, profiles }: VacancyListProps) {
  const router = useRouter();
  const [deleteDialogOpen, setDeleteDialogOpen] = useState(false);
  const [selectedVacancyId, setSelectedVacancyId] = useState<string | null>(null);

  const selectedVacancy = selectedVacancyId ? vacancies.find((v) => v.id === selectedVacancyId) : null;
  const selectedProfile = selectedVacancy ? profiles.get(selectedVacancy.profileId) : null;

  const handleDeleteClick = (vacancyId: string) => {
    setSelectedVacancyId(vacancyId);
    setDeleteDialogOpen(true);
  };

  const handleConfirmDelete = async () => {
    if (selectedVacancyId) {
      await deleteVacancy(selectedVacancyId);
      setDeleteDialogOpen(false);
      setSelectedVacancyId(null);
      router.refresh();
    }
  };

  const handleAdvanceStatus = async (vacancyId: string) => {
    await advanceVacancyStatus(vacancyId);
    router.refresh();
  };

  if (vacancies.length === 0) {
    return (
      <div className="text-center py-12">
        <h3 className="text-title-md font-medium text-on-surface mb-2">
          Nenhuma vaga aberta ainda
        </h3>
        <p className="text-body-md text-on-surface/60 mb-6">
          Abra uma nova vaga selecionando um perfil. Cada vaga rastreia o ciclo de vida da requisição.
        </p>
        <Link href="/vacancies/new">
          <Button>Abrir primeira vaga</Button>
        </Link>
      </div>
    );
  }

  // Sort by openedAt descending (most recent first, per D-15)
  const sortedVacancies = [...vacancies].sort(
    (a, b) => new Date(b.openedAt).getTime() - new Date(a.openedAt).getTime()
  );

  return (
    <>
      <div className="flex flex-col gap-0 border border-surface-container-lowest/30">
        {sortedVacancies.map((vacancy) => {
          const profile = profiles.get(vacancy.profileId);
          const vacancyTitle = profile?.title ?? "(Perfil não encontrado)";

          return (
            <div
              key={vacancy.id}
              className="flex items-center justify-between py-4 px-4 gap-4 border-b border-surface-container-lowest/20 last:border-b-0"
            >
              {/* Left: Title, Profile, Badge */}
              <div className="flex-1 min-w-0 flex items-center gap-4">
                <div className="flex-1">
                  <p className="text-body-md font-medium text-on-surface truncate">
                    {vacancyTitle}
                  </p>
                  <div className="flex items-center gap-2 text-body-sm text-on-surface/70">
                    <span>
                      Qty: {vacancy.quantity} • {vacancy.costCenter || "—"}
                    </span>
                  </div>
                </div>
                <Badge variant={STATUS_BADGE_VARIANT[vacancy.status]}>
                  {vacancy.status}
                </Badge>
              </div>

              {/* Right: Date + Actions */}
              <div className="flex items-center gap-2 shrink-0">
                <span className="text-label-sm uppercase text-on-surface/70">
                  {new Date(vacancy.openedAt).toLocaleDateString("pt-BR")}
                </span>

                {/* Status advance button */}
                {vacancy.status !== "Encerrada" && (
                  <Button
                    size="icon"
                    variant="ghost"
                    onClick={() => handleAdvanceStatus(vacancy.id)}
                    title="Avançar status"
                    className="w-8 h-8"
                  >
                    <ChevronRight className="w-4 h-4" />
                  </Button>
                )}

                {/* Edit button */}
                <Link href={`/vacancies/${vacancy.id}/edit`}>
                  <Button size="icon" variant="ghost" title="Editar vaga" className="w-8 h-8">
                    <Pencil className="w-4 h-4" />
                  </Button>
                </Link>

                {/* Delete button */}
                <Button
                  size="icon"
                  variant="ghost"
                  onClick={() => handleDeleteClick(vacancy.id)}
                  title="Excluir vaga"
                  className="w-8 h-8"
                >
```
