Executar o script de diagnóstico abaixo para extrair os endereços de célula com valores preenchidos nos três exemplos. O script extrai o XML do sheet1 de cada arquivo e exibe o mapeamento `<r de célula> → <valor>`:

```bash
node -e "
const AdmZip = require('adm-zip');
const path = require('path');

const examples = [
  'data/examples/cientista-dados.xlsx',
  'data/examples/dev-frontend-pleno.xlsx',
  'data/examples/dev-java-pleno.xlsx',
];

for (const examplePath of examples) {
  console.log('\n\n=== ' + examplePath + ' ===');
  const zip = new AdmZip(examplePath);
  const sheetEntry = zip.getEntry('xl/worksheets/sheet1.xml');
  if (!sheetEntry) { console.log('sheet1.xml não encontrado'); continue; }
  const xml = sheetEntry.getData().toString('utf-8');

  // Extrair shared strings (sst) para resolver referências t='s'
  const sstEntry = zip.getEntry('xl/sharedStrings.xml');
  const sharedStrings = [];
  if (sstEntry) {
    const sstXml = sstEntry.getData().toString('utf-8');
    const siMatches = sstXml.matchAll(/<si[^>]*>(.*?)<\/si>/gs);
    for (const m of siMatches) {
      const text = m[1].replace(/<[^>]+>/g, '').trim();
      sharedStrings.push(text);
    }
  }

  // Extrair todas as células com valor não vazio
  const cellMatches = xml.matchAll(/<c r=\"([A-Z]+\d+)\"[^>]*(?:t=\"([^\"]*)\")[^>]*>(?:<v>(\d+)<\/v>|<is><t>([^<]*)<\/t><\/is>)/g);
  const results = [];
  for (const m of cellMatches) {
    const addr = m[1];
    const type = m[2];
    const vVal = m[3];
    const isVal = m[4];
    let value;
    if (type === 's' && vVal !== undefined) {
      value = sharedStrings[parseInt(vVal)] || '(shared#' + vVal + ')';
    } else if (isVal !== undefined) {
      value = isVal;
    } else if (vVal !== undefined) {
      value = vVal;
    }
    if (value && value.trim()) {
      results.push(addr + ' = ' + JSON.stringify(value.trim().substring(0, 80)));
    }
  }

  // Também capturar células sem atributo t= mas com <v>
  const noTypeCells = xml.matchAll(/<c r=\"([A-Z]+\d+)\"[^>]*>(?!<is>)<v>([^<]+)<\/v>/g);
  for (const m of noTypeCells) {
    results.push(m[1] + ' = (number/bool) ' + m[2]);
  }

  results.sort().forEach(r => console.log(r));
}
"
```
